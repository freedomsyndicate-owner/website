function render(trades) { const h = document.getElementById('hist'); h.innerHTML = ''; let totalP = 0, win = 0; // Sort dari yang terbaru di atas const sorted = trades.sort((a,b) => new Date(b.date+' '+b.time) - new Date(a.date+' '+a.time)); sorted.forEach(t => { totalP += (t.pnl || 0); if(t.res === 'TP') win++; const card = document.createElement('div'); card.className = 'history-card'; // Gabungkan array models jadi teks (contoh: QM, FVG) const modelsText = t.models ? t.models.join(', ') : '-'; card.innerHTML = `
×
${t.date} ${t.time} | ${t.pair} (${t.res})
Emotion: ${t.emotion || '-'} | Model: ${modelsText}
Note: ${t.reason || 'No notes'}
${t.bef ? `` : '
No Before
'} ${t.aft ? `` : '
No After
'}
P&L: $${t.pnl} RR: ${t.rr}
`; h.appendChild(card); }); // Update statistik di tab Monitor const wr = trades.length > 0 ? Math.round((win / trades.length) * 100) : 0; document.getElementById('st-pnl').innerText = `$ ${totalP.toFixed(2)}`; document.getElementById('st-pnl').style.color = totalP >= 0 ? 'var(--green)' : 'var(--red)'; document.getElementById('st-wr').innerText = `${wr}%`; renderCalendar(trades); } function renderCalendar(trades) { const cal = document.getElementById('cal'); cal.innerHTML = ''; const grouped = {}; trades.forEach(t => { grouped[t.date] = (grouped[t.date] || 0) + t.pnl; }); // Render 30 hari terakhir di kalender for(let i=29; i>=0; i--) { const d = new Date(); d.setDate(d.getDate() - i); const ds = d.toISOString().split('T')[0]; const pnl = grouped[ds] || 0; const day = document.createElement('div'); day.className = `cal-day ${pnl > 0 ? 'profit' : pnl < 0 ? 'loss' : ''}`; day.innerHTML = `${d.getDate()}${pnl !== 0 ? (pnl > 0 ? '+' : '') + pnl.toFixed(0) : ''}`; cal.appendChild(day); } } // Jalankan inisialisasi awal window.onload = () => { const now = new Date(); document.getElementById('date').value = now.toISOString().split('T')[0]; document.getElementById('time').value = now.toTimeString().slice(0,5); };
